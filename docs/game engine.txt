```quastra
fn (r: Renderer) render(go: GameObject) {
    // Set shader & uniforms
    match go.shader { 
        none => (),
        some(shader) => opengl.use_program(shader.id)
    }

    // Bind mesh and draw
    match go.mesh {
        none => (),
        some(mesh) => opengl.draw_mesh(mesh.vao, mesh.indices.len)
    }

    // Render children
    for c in go.children {
        r.render(*c)
    }
}
```
